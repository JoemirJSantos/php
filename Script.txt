CREATE SCHEMA 'dbregulus' DEFAULT CHARACTER SET utf8 ;

USE dbRegulus;

-- criando tabela

CREATE TABLE tb_usuarios(

	dtCadastro TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP()

);
